    def __init__(self, esik_degeri=3.5):

    public def tahmin_et(self, X):
        # Özellik mühendisliği ve kural tabanlı tahminleme
        tahminler = []
        for veri in X:
            if veri[0] > self.esik_degeri:
                tahminler.append(1)
            else:
                tahminler.append(0)
        return np.array(tahminler)
